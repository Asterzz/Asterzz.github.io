def stddev(arr, window, axis=1):
    """
    ``sdev`` , return the time series stddev of ndarray data(index-time;cols-symbols)

    Parameters
    ----------
    arr : ndarray
        index represent time cols represent symbols
        
    window : int or float
        will take int(abs(window)) as window size if abs(window)<3 take 3
        
    dd : default 0
        ddof of stddev calculation
        
    axis : default 1
        axis of calculation, 1 for row, 0 for cols

    Returns
    -------
    stddev
        An numpy ndarray of the same shape,the time series stddev of input

    """

    return sdev.stddev(arr, window, axis=axis)

def ts_returns(arr, window, MODE=1, axis=1):
    """
    ``sdev`` , 
    MODE == 1:
    return (x - tsdelay(x, n)) / tsdelay(x, n)
    MODE == 2:
    return (x = tsdelay(x, n)) / ((x + tsdelay(x,n)) /2)

    Parameters
    ----------
    arr : ndarray
        index represent time cols represent symbols
        
    window : int or float
        will take int(abs(window)) as window size; if abs(window)<1 will take 1
    
    MODE : int
        default 1, 1 or 2

    axis : default 1
        axis of calculation, 1 for row, 0 for cols

    Returns
    -------
    ts_returns    
        An numpy ndarray of the same shape, (x - delay(x, window)) / delay(x, window) of input

    """

    return sdev.ts_returns(arr, window, MODE, axis=axis)

def ts_regression(arr1, arr2, window, RETTYPE, axis=1):
    """
    ``sdev``  , Perform time series regression.

    Given linear regression y = b * x + a, output is (based on RETTYPE):
        0 = Y - EY
        1 = a
        2 = b
        3 = Ey
        4 = square
        5 = adjust r square

    Parameters
    ----------
    arr1 : ndarray
        Index represents time, columns represent symbols.
        
    arr2 : ndarray
        Index represents time, columns represent symbols.
        
    window : int or float
        Will take int(abs(window)) as window size; if abs(window) < 1, will take 1.
    
    RETTYPE : int
        Output type:
            0 = Y - EY
            1 = a
            2 = b
            3 = Ey
            4 = square
            5 = adjust r square
        
    axis : int, optional
        Axis of calculation, 1 for row, 0 for cols. Default is 1.

    Returns
    -------
    ndarray
        A numpy ndarray of the same shape, the time series regression of input1 and input2.

    """
    return sdev.ts_regression(arr1, arr2, window, RETTYPE, axis=axis)
